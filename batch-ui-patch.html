import concurrent.futures import zipfile from pathlib import Path from typing import List, Dict, Tuple import cv2 import numpy as np from skimage.metrics import structural_similarity as ssim from PIL import Image def create_batch_executor(max_workers: int = 4): """ThreadPoolExecutor for parallel image processing.""" return concurrent.futures.ThreadPoolExecutor(max_workers=max_workers) def validate_image(input_path: str, max_size_mb: int = 20) -> Tuple[bool, str]: """Validate image by extension, size, and actual file content.""" try: path = Path(input_path) allowed_extensions = {".jpg", ".jpeg", ".png", ".webp"} ext = path.suffix.lower() if ext and ext not in allowed_extensions: return False, "Unsupported format (JPG/JPEG/PNG/WEBP only)" if not path.exists(): return False, "File not found" size_mb = path.stat().st_size / (1024 * 1024) if size_mb > max_size_mb: return False, f"File too large ({size_mb:.1f} MB > {max_size_mb} MB)" try: with Image.open(path) as img: img.verify() except Exception: img = cv2.imread(str(path)) if img is None or img.size == 0: return False, "Invalid or corrupt image" return True, "" except Exception as e: return False, str(e) def compute_metrics(original: np.ndarray, enhanced: np.ndarray) -> Dict[str, float]: """Compute PSNR and SSIM metrics between original and enhanced.""" h, w = original.shape[:2] enhanced_resized = cv2.resize(enhanced, (w, h), interpolation=cv2.INTER_AREA) original_float = original.astype(np.float32) enhanced_float = enhanced_resized.astype(np.float32) mse = np.mean((original_float - enhanced_float) ** 2) if mse == 0: psnr = float("inf") else: psnr = 20 * np.log10(255.0 / np.sqrt(mse)) gray_orig = cv2.cvtColor(original, cv2.COLOR_BGR2GRAY) gray_enh = cv2.cvtColor(enhanced_resized, cv2.COLOR_BGR2GRAY) data_range = float(gray_enh.max() - gray_enh.min()) if data_range == 0: data_range = 255.0 ssim_val = ssim(gray_orig, gray_enh, data_range=data_range) return { "psnr": round(psnr, 2) if np.isfinite(psnr) else 999.0, "ssim": round(float(ssim_val), 4), } def create_zip_batch(batch_id: str, output_paths: List[Path], original_names: List[str]) -> str: """Create ZIP of batch results.""" output_dir = Path("outputs") output_dir.mkdir(exist_ok=True) zip_path = output_dir / f"{batch_id}_batch_results.zip" with zipfile.ZipFile(zip_path, "w", zipfile.ZIP_DEFLATED) as zf: for out_path, orig_name in zip(output_paths, original_names): arcname = f"{Path(orig_name).stem}_enhanced.png" zf.write(out_path, arcname) return str(zip_path)
